1. Char replace in String

This code replace a charcter in String

import java.util.Scanner;
class Demo{
public String change(String str,char char1,int index){
String str2;
if(index==0){
str2=char1+str.substring(1);
}else {
str2=str.substring(0,index)+char1+str.substring(index+1);
}
return str2;
}
}
class Main{
public static void main(String args[]){
Scanner input=new Scanner(System.in);
Demo obj=new Demo();
System.out.println("Enter String");
String str=input.nextLine();
System.out.println("inter index");
int index= input.nextInt();
if(index>=str.length()){
System.out.println("Index of Bound");
}else{
System.out.println("Enter char for replace");
char char1=input.next().charAt(0);
String str2=obj.change(str,char1,index);
System.out.println("Before:- "+str);
System.out.println("After :- "+str2);
}
}
}

2. Prime Or Not

This code check the number is prime or not

import java.util.Scanner;
class Main{
public static void main(String args[]){
Scanner input =new Scanner(System.in);
System.out.println("Enter Your Number ");
int number=input.nextInt();
Demo obj=new Demo();
int recive =obj.prime(number);
if(recive==1){
System.out.println("Number is prime ");
}else{
System.out.println("Number is Not Prine");
}
}
}
class Demo{
public int prime;
Demo(){
prime=0;
}
public int prime(int number){
for(int i=2;i<=number;i++){
if(number%i==0){
prime++;
}
}
return prime;
}
}
